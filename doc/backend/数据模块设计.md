模块一：核心基建与患者 (Base & Patient)

patients
  id (UUID, PK)
  name, gender, age, phone, id_number (VARCHAR)
  insurance_card_no (VARCHAR, UNIQUE)
  status (ENUM: checked-in, waiting, consulting, pending-treatment, treating, completed)
  created_at, updated_at (TIMESTAMPTZ)

contraindications (禁忌症字典表 - 预置数据)
  id (UUID, PK)
  code (VARCHAR, UNIQUE) - 禁忌症编码
  name (VARCHAR)
  pinyin (VARCHAR) - 拼音全拼（搜索用）
  pinyin_initial (VARCHAR) - 拼音首字母（快捷搜索用）
  category (VARCHAR) - 分类

scale_templates (量表模板)
  id (UUID, PK)
  name (VARCHAR)
  description (TEXT)

scale_questions (量表题目)
  id (UUID, PK)
  template_id (UUID, FK → scale_templates)
  text (TEXT) - 题目内容
  type (ENUM: single-choice, multi-choice, slider, text)
  required (BOOLEAN)
  options (JSONB) - 选项列表 [{value, label, score}]
  slider_config (JSONB) - 滑块配置 {min, max, step}，仅 slider 类型用


模块二：疗愈内容库 (从 CSV 映射而来)

therapy_projects (疗愈项目)
  id (UUID, PK)
  region (VARCHAR) - 睡眠区/情志区/运动区
  name (VARCHAR)
  mechanism (TEXT) - 机制说明
  guidance_script (TEXT) - 引导语原文
  bpm (INTEGER) - 速度
  mood, energy_level (VARCHAR) - 情绪、唤醒度
  has_guidance (BOOLEAN) - 有无引导语
  has_scenario (BOOLEAN) - 有无情景
  target_audience (VARCHAR) - 适合人群
  notes (TEXT) - 备注
  is_internal (BOOLEAN) - 内部数据过滤

therapy_packages (疗愈套餐)
  id (UUID, PK)
  name (VARCHAR) - 如"失眠调理套餐"
  target_audience (TEXT) - 适用人群
  matched_symptoms (TEXT) - 匹配症状

therapy_package_items (多对多关联)
  package_id (UUID, FK → therapy_packages)
  project_id (UUID, FK → therapy_projects)
  sort_order (INTEGER) - 项目在套餐中的顺序


模块三：业务流转核心 (The Hub)

consultations (就诊主记录 - 串联分诊、医生、治疗)
  id (UUID, PK)
  patient_id (UUID, FK → patients)
  status (VARCHAR)
  created_at (TIMESTAMPTZ)

consultation_contraindications (就诊-禁忌症关联表)
  consultation_id (UUID, FK → consultations)
  contraindication_id (UUID, FK → contraindications)
  PK: (consultation_id, contraindication_id)

vital_signs (体征数据)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations)
  stage (ENUM: pre-treatment, post-treatment)
  systolic_bp, diastolic_bp, heart_rate (INTEGER)
  recorded_at (TIMESTAMPTZ)
  recorded_by (VARCHAR)

scale_results (量表结果)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations)
  stage (ENUM: pre, post)
  template_id (UUID, FK → scale_templates)
  answers (JSONB) - 答卷 {questionId: value}
  total_score (INTEGER)
  submitted_at (TIMESTAMPTZ)

ai_suggestions (AI 处方建议记录)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations, UNIQUE)
  model (VARCHAR) - 调用的模型名，如 qwen3.5-plus
  prompt_snapshot (TEXT) - 发送给 AI 的完整 prompt（审计用）
  raw_response (TEXT) - AI 原始返回
  herbs (JSONB) - 解析后的药材列表 [{name, dosage, unit, reason}]
  usage (TEXT) - 用法
  notes (TEXT) - 注意事项
  confidence (DECIMAL)
  generated_at (TIMESTAMPTZ)

prescriptions (处方)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations, UNIQUE)
  therapy_package_id (UUID, FK → therapy_packages) - 关联疗愈套餐
  meta (JSONB) - {route, usage, frequency, dosage, orderType, department, doses}
  herbs (JSONB) - [{name, dosage, unit, note}]
  total_amount (DECIMAL)
  created_at (TIMESTAMPTZ)


模块四：队列与治疗执行

queue_items (候诊/治疗队列)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations)
  patient_id (UUID, FK → patients)
  queue_type (ENUM: waiting, treatment) - 候诊队列 or 治疗队列
  queue_number (INTEGER)
  status (ENUM: waiting, in-progress, completed)
  enqueued_at (TIMESTAMPTZ)

treatment_records (治疗执行记录 - 每个房间一条)
  id (UUID, PK)
  consultation_id (UUID, FK → consultations)  // 去掉 UNIQUE，改为 1:N
  region (VARCHAR) - 本次治疗的房间
  start_time (TIMESTAMPTZ)
  end_time (TIMESTAMPTZ)
  duration (INTEGER) - 秒
  created_at (TIMESTAMPTZ)

prescription_steps (处方执行步骤 - 多房间流转核心)
  id (UUID, PK)
  prescription_id (UUID, FK → prescriptions)
  project_id (UUID, FK → therapy_projects)
  region (VARCHAR) - 睡眠区/情志区/运动疗愈区（冗余，方便终端按房间过滤）
  sort_order (INTEGER) - 项目在套餐中的顺序
  status (ENUM: pending, in-progress, completed, skipped)
  started_at (TIMESTAMPTZ, NULLABLE)
  completed_at (TIMESTAMPTZ, NULLABLE)
  treatment_record_id (UUID, FK → treatment_records, NULLABLE)


模块五：系统配置

system_config (系统配置 - API Key 等可热更换配置)
  id (UUID, PK)
  key (VARCHAR, UNIQUE) - 如 'DASHSCOPE_API_KEY', 'AI_MODEL_NAME'
  value (TEXT) - 敏感值加密存储
  description (TEXT)
  updated_at (TIMESTAMPTZ)
