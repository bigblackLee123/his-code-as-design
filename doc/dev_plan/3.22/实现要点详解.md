export function ContraindicationInput({ patientId, onItemsChange }) {
  const [allItems, setAllItems] = useState<Contraindication[]>([]);
  const [keyword, setKeyword] = useState("");
  const [items, setItems] = useState<Contraindication[]>([]);

  useEffect(() => {
    contraindicationService.getAll().then(setAllItems);
  }, []);

  // 按 category 分组
  const grouped = useMemo(() => {
    const g: Record<string, Contraindication[]> = {};
    allItems.forEach(item => {
      (g[item.category] ??= []).push(item);
    });
    return g;
  }, [allItems]);

  // 过滤
  const filtered = useMemo(() => {
    if (!keyword) return grouped;
    const k = keyword.toLowerCase();
    const result: Record<string, Contraindication[]> = {};
    Object.entries(grouped).forEach(([cat, items]) => {
      const matched = items.filter(i => 
        i.name.toLowerCase().includes(k) || 
        i.pinyinInitial.includes(k)
      );
      if (matched.length > 0) result[cat] = matched;
    });
    return result;
  }, [grouped, keyword]);

  return (
    <div>
      <Input 
        placeholder="搜索禁忌症..."
        value={keyword}
        onChange={e => setKeyword(e.target.value)}
      />
      {Object.entries(filtered).map(([category, items]) => (
        <div key={category}>
          <h4>{category}</h4>
          {items.map(item => (
            <div key={item.code}>
              <Checkbox
                checked={items.some(i => i.code === item.code)}
                onChange={checked => {
                  if (checked) handleAdd(item);
                  else handleRemove(item.code);
                }}
              />
              <label>{item.name}</label>
            </div>
          ))}
        </div>
      ))}
    </div>
  );
}

          )}
        </div>
      ))}
    </div>
  );
}

  if (req.method === "OPTIONS") {

      supabase,
      patient.contraindications,
    );

    const userMessage = buildUserMessage(patient, candidates);

    // 使用流式响应
    const encoder = new TextEncoder();
    const readable = new ReadableStream({
      async start(controller) {
        try {
          const rawText = aiCfg.appId
            ? await callBailianApp(aiCfg, userMessage)
            : await callDirectModel(aiCfg, userMessage);

          // 逐字发送 reason
          const jsonMatch = rawText.match(/\{[\s\S]*\}/);
          if (!jsonMatch) {
            throw new Error(`AI 返回格式异常: ${rawText.slice(0, 200)}`);
          }

          const parsed = JSON.parse(jsonMatch[0]);
          const reason = parsed.reason ?? "";

          // 流式发送 reason 的每个字符
          for (const char of reason) {
            const chunk = JSON.stringify({ 
              type: "reason_chunk", 
              data: char 
            });
            controller.enqueue(encoder.encode(chunk + "\n"));
            // 模拟打字延迟
            await new Promise(r => setTimeout(r, 20));
          }

          // 最后发送完整结果
          const result = {
            id: `AI-${Date.now()}`,
            projectIds: parsed.projectIds ?? [],
            projectNames: parsed.projectNames ?? [],
            reason: reason,
            confidence: parsed.confidence ?? 0.5,
            generatedAt: new Date().toISOString(),
          };

          const finalChunk = JSON.stringify({ 
            type: "complete", 
            data: result 
          });
          controller.enqueue(encoder.encode(finalChunk + "\n"));
          controller.close();
        } catch (err) {
          const msg = err instanceof Error ? err.message : String(err);
          controller.enqueue(encoder.encode(
            JSON.stringify({ type: "error", data: msg }) + "\n"
          ));
          controller.close();
        }
      },
    });

    return new Response(readable, {
      headers: {
        ...corsHeaders,
        "Content-Type": "application/x-ndjson",
        "Transfer-Encoding": "chunked",
      },
    });
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    console.error("[ai-therapy-suggestion]", msg);
    return new Response(JSON.stringify({ error: msg }), {
      status: 500,
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  }
});
